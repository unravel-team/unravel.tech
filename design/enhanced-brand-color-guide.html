CSS Architecture

/* Recommended CSS Structure */
/* 1. Define in :root for light mode */
:root {
  --color-primary: #0E9FBC;
  --color-text: var(--color-neutral-900);
  --color-bg: var(--color-neutral-000);
}

/* 2. Override for dark mode */
[data-theme="dark"] {
  --color-primary: #4FC3DB;
  --color-text: #C9D1D9;
  --color-bg: #0D1117;
}

/* 3. Use semantic tokens in components */
.button-primary {
  background: var(--color-primary);
  color: var(--color-text-on-primary);
}
